```
function uniTotal(str) {
  if (str === undefined) {
    return null;
  } else if (str.length === 0) {
    return 0;
  }
  
  let res = 0;
  
  for (let i = 0; i < str.length; i++) {
    res += str.charCodeAt(i);
  }
  
  return res;
}
```
